選擇排序法 Selection Sort

```javascript
function s(arr) {
  if (!arr.length) return arr;
  let min = {};
  for (let i = 0; i < arr.length; i++) {
    min.index = i;
    min.val = arr[i];
    for (let j = i; j < arr.length; j++) {
      if (arr[j] < min.val) {
        min.val = arr[j];
        min.index = j;
      }
    }
    [arr[i], arr[min.index]] = [min.val, arr[i]];
  }

  return arr;
}

const arr = [2, 5, 3, 1, 10];
const arr2 = [69,81,30,38,9,2,47,61,32,79];
console.log(s(arr));
console.log(s(arr2));

```
