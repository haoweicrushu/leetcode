function f(str) {
    let index = 0;
    const m = Array.from(str).reduce((prev, curr) => {
      if (prev[curr]) {
        prev[curr].count++;
      } else {
        prev[curr] = {
          count: 1,
          index,
        };
      }
      index++;
      return prev;
    }, {});
    return Object.keys(m)
      .map(key => ({ ...m[key], char: key }))
      .sort((a, b) => b.count - a.count)
      .map(b => {
        let s = '';
        while (b.count > 0) {
          s += b.char;
          b.count--;
        }
        return s;
      })
      .join('');
  }
  
  console.log(f('aaaccAAAA'));
